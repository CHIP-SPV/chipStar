412 bool init(unsigned int maxThreadCount)
413 {
414 int rc = 0;
416 _count = 0xFFFFFFFF;
417 _maxThreads = maxThreadCount;
418#ifdef _WIN32
419 _nLockCount = 0;
420 // Initialize the critical section that protects access to
421 // the wait set. There is no way to check for errors here.
422 InitializeCriticalSection(&_critsecWaitSetProtection);
423 // Initialize the critical section that provides synchronization
424 // to the client. There is no way to check for errors here.
425 InitializeCriticalSection(&_condVarLock);
426#else
427 pthread_mutex_init(&_condVarLock, NULL);
428 PRINT_ERROR_MSG(rc, "Failed to initialize condition variable lock");
429 rc = pthread_cond_init(&_condVar, NULL);
430 PRINT_ERROR_MSG(rc, "Failed to Initialize condition variable");
431#endif
432 if(rc != 0)
433 {
434 return false;
435 }
436 return true;
437 }

442 bool destroy()
443 {
444 int rc = 0;
445#ifdef _WIN32
446 // Uninitialize critical sections. Win32 allows no error checking
447 // here.
448 DeleteCriticalSection(&_critsecWaitSetProtection);
449 DeleteCriticalSection(&_condVarLock);
450 // Destroying this thing while threads are waiting is a client
451 // programmer mistake.
452 assert( _deqWaitSet.empty() );
453#else
455 pthread_mutex_destroy(&_condVarLock);
456 PRINT_ERROR_MSG(rc, "Failed to destroy condition variable lock");
458 rc = pthread_cond_destroy(&_condVar);

471 void syncThreads()
472 {
474 beginSynchronized();
475 int rc = 0;
477 if(_count == 0xFFFFFFFF)
478 {
479 _count = 0;
480 }
481 else
482 {
483 _count++;
484 }
485 if(_count >= _maxThreads - 1)
486 {
488 _count = 0xFFFFFFFF;
490#ifdef _WIN32
491 rc = broadcast();
492 if(rc == 0)
493 {
494 printf("Problem while calling broadcast\n");
495 }
496#else
497 rc = pthread_cond_broadcast(&_condVar);
498 PRINT_ERROR_MSG(rc, "Problem while calling pthread_cond_broadcast()");
499#endif // _WIN32
500 }
501 else
502 {
504#ifdef _WIN32
505 wait();
506#else
507 if(_count < _maxThreads - 1)
508 {
509 rc = pthread_cond_wait(&_condVar, &_condVarLock);
510 PRINT_ERROR_MSG(rc, "Problem while calling pthread_cond_wait()");
511 }
512#endif // _WIN32
513 }
515 endSynchronized();
516 }

524 void beginSynchronized()
525 {
526#ifdef _WIN32
527 // Acquire lock. Win32 allows no error checking here.
528 EnterCriticalSection(&_condVarLock);
529 // Record the lock acquisition for proper release in Wait().
530 ++_nLockCount;
531#else
532 pthread_mutex_lock(&_condVarLock);
533#endif // _WIN32
534 }

539 int endSynchronized()
540 {
541#ifdef _WIN32
542 if(! _lockHeldByCallingThread())
543 {
544 ::SetLastError(ERROR_INVALID_FUNCTION); // for the lack of better...
545 return 0;
546 }
547 // Record the lock release for proper release in Wait().
548 --_nLockCount;
549 // Release lock. Win32 allows no error checking here.
550 ::LeaveCriticalSection(&_condVarLock);
551#else
552 pthread_mutex_unlock(&_condVarLock);
553#endif // _WIN32
554 return 1;
555 }

557#ifdef _WIN32

562 DWORD wait(DWORD dwMillisecondsTimeout = INFINITE, BOOL bAlertable = FALSE)
563 {
564 if(! _lockHeldByCallingThread())
565 {
566 ::SetLastError(ERROR_INVALID_FUNCTION); // for the lack of better...
567 return WAIT_FAILED;
568 }
569 // Enter a new event handle into the wait set.
570 HANDLE hWaitEvent = _push();
571 if(NULL == hWaitEvent)
572 {
573 return WAIT_FAILED;
574 }
575 // Store the current lock count for re-acquisition.
576 int nThisThreadsLockCount = _nLockCount;
577 _nLockCount = 0;
578 // Release the synchronization lock the appropriate number of times.
579 // Win32 allows no error checking here.
580 for(int i = 0; i < nThisThreadsLockCount; ++i)
581 {
582 LeaveCriticalSection(&_condVarLock);
583 }
584 // NOTE: Conceptually, releasing the lock and entering the wait
585 // state is done in one atomic step. Technically, that is not
586 // true here, because we first leave the critical section and
587 // then, in a separate line of code, call WaitForSingleObjectEx.
588 // The reason why this code is correct is that our thread is placed
589 // in the wait set *before* the lock is released. Therefore, if
590 // we get preempted right here and another thread notifies us, then
591 // that notification will *not* be missed: the wait operation below
592 // will find the event signalled.
593 // Wait for the event to become signalled.
594 DWORD dwWaitResult = ::WaitForSingleObjectEx(hWaitEvent, dwMillisecondsTimeout,
595 bAlertable);
596 // If the wait failed, store the last error because it will get
597 // overwritten when acquiring the lock.
598 DWORD dwLastError = 0;
599 if(WAIT_FAILED == dwWaitResult)
600 {
601 dwLastError = ::GetLastError();
602 }
603 // Acquire the synchronization lock the appropriate number of times.
604 // Win32 allows no error checking here.
605 for(int j = 0; j < nThisThreadsLockCount; ++j)
606 {
607 EnterCriticalSection(&_condVarLock);
608 }
609 // Restore lock count.
610 _nLockCount = nThisThreadsLockCount;
611 // Close event handle
612 if(! CloseHandle(hWaitEvent))
613 {
614 return WAIT_FAILED;
615 }
616 if(WAIT_FAILED == dwWaitResult)
617 {
618 ::SetLastError(dwLastError);
619 }
620 return dwWaitResult;
621 }

626 BOOL broadcast()
627 {
628 // Signal all events on the deque, then clear it. Win32 allows no
629 // error checking on entering and leaving the critical section.
630 EnterCriticalSection(&_critsecWaitSetProtection);
631 std::deque<HANDLE>::const_iterator it_run = _deqWaitSet.begin();
632 std::deque<HANDLE>::const_iterator it_end = _deqWaitSet.end();
633 for( ; it_run < it_end; ++it_run)
634 {
635 if(! SetEvent(*it_run))
636 {
637 return FALSE;
638 }
639 }
640 _deqWaitSet.clear();
641 LeaveCriticalSection(&_critsecWaitSetProtection);
642 return TRUE;
643 }
644
651 HANDLE _push()
652 {
653 // Create the new event.
654 HANDLE hWaitEvent = ::CreateEvent(NULL, // no security
655 FALSE, // auto-reset event
656 FALSE, // initially unsignalled
657 NULL); // string name
658 if(NULL == hWaitEvent)
659 {
660 return NULL;
661 }
662 // Push the handle on the deque.
663 EnterCriticalSection(&_critsecWaitSetProtection);
664 _deqWaitSet.push_back(hWaitEvent);
665 LeaveCriticalSection(&_critsecWaitSetProtection);
666 return hWaitEvent;
667 }
668
673 HANDLE _pop()
674 {
675 // Pop the first handle off the deque.
676 EnterCriticalSection(&_critsecWaitSetProtection);
677 HANDLE hWaitEvent = NULL;
678 if(0 != _deqWaitSet.size())
679 {
680 hWaitEvent = _deqWaitSet.front();
681 _deqWaitSet.pop_front();
682 }
683 LeaveCriticalSection(&_critsecWaitSetProtection);
684 return hWaitEvent;
685 }
686
690 BOOL _lockHeldByCallingThread()
691 {
692 BOOL bTryLockResult = TryEnterCriticalSection(&_condVarLock);
693 // If we didn't get the lock, someone else has it.
694 if(!bTryLockResult)
695 {
696 return FALSE;
697 }
698 // If we got the lock, but the lock count is zero, then nobody had it.
699 if(0 == _nLockCount)
700 {
701 assert(bTryLockResult);
702 LeaveCriticalSection(&_condVarLock);
703 return FALSE;
704 }
705 // Release lock once. NOTE: we still have it after this release.
706 // Win32 allows no error checking here.
707 assert(bTryLockResult && 0 < _nLockCount);
708 LeaveCriticalSection(&_condVarLock);
709 return TRUE;
710 }
