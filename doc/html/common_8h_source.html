105// Log level enum
106namespace level {
107enum level_enum {
108 trace = SPDLOG_LEVEL_TRACE,
109 debug = SPDLOG_LEVEL_DEBUG,
110 info = SPDLOG_LEVEL_INFO,
111 warn = SPDLOG_LEVEL_WARN,
112 err = SPDLOG_LEVEL_ERROR,
113 critical = SPDLOG_LEVEL_CRITICAL,
114 off = SPDLOG_LEVEL_OFF,
115};
116
117#if !defined(SPDLOG_LEVEL_NAMES)
118#define SPDLOG_LEVEL_NAMES \
119 { "trace", "debug", "info", "warning", "error", "critical", "off" }
120#endif
121
122static string_view_t level_string_views[] SPDLOG_LEVEL_NAMES;
123static const char *short_level_names[]{"T", "D", "I", "W", "E", "C", "O"};
124
125inline string_view_t &to_string_view(spdlog::level::level_enum l)
126 SPDLOG_NOEXCEPT {
127 return level_string_views[l];
128}
129
130inline const char *to_short_c_str(spdlog::level::level_enum l) SPDLOG_NOEXCEPT {
131 return short_level_names[l];
132}
133
134inline spdlog::level::level_enum from_str(const std::string &name)
135 SPDLOG_NOEXCEPT {
136 int level = 0;
137 for (const auto &level_str : level_string_views) {
138 if (level_str == name) {
139 return static_cast<level::level_enum>(level);
140 }
141 level++;
142 }
143 return level::off;
144}

161class spdlog_ex : public std::exception {
162 public:
163 explicit spdlog_ex(std::string msg) : msg_(std::move(msg)) {}
164
165 spdlog_ex(const std::string &msg, int last_errno) {
166 fmt::memory_buffer outbuf;
167 fmt::format_system_error(outbuf, last_errno, msg);
168 msg_ = fmt::to_string(outbuf);
169 }
170
171 const char *what() const SPDLOG_NOEXCEPT override { return msg_.c_str(); }
172
173 private:
174 std::string msg_;
175};
