37// Get Free Memory from the system
38static size_t getMemoryAmount() {
39#ifdef __linux__
40 struct sysinfo info{};
41 sysinfo(&info);
42 return info.freeram / (1024 * 1024); // MB
43#elif defined(_WIN32)
44 MEMORYSTATUSEX statex;
45 statex.dwLength = sizeof(statex);
46 GlobalMemoryStatusEx(&statex);
47 return (statex.ullAvailPhys / (1024 * 1024)); // MB
48#endif
49}

51static size_t getHostThreadCount(const size_t memPerThread,
52 const size_t maxThreads) {
53 if (memPerThread == 0) return 0;
54 auto memAmount = getMemoryAmount();
55 const auto processor_count = std::thread::hardware_concurrency();
56 if (processor_count == 0 || memAmount == 0) return 0;
57 size_t thread_count = 0;
58 if ((processor_count * memPerThread) < memAmount)
59 thread_count = processor_count;
60 else
61 thread_count = reinterpret_cast<size_t>(memAmount / memPerThread);
62 if (maxThreads > 0) {
63 return (thread_count > maxThreads) ? maxThreads : thread_count;
64 }
65 return thread_count;
66}
