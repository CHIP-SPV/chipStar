43namespace hip {
44class SpawnProc {
45 std::string exeName;
46 std::string resultStr;
47 std::string tmpFileName;
48 bool captureOutput;
49
50 std::string getRandomString(size_t len = 6) {
51 std::random_device dev;
52 std::mt19937 rng(dev());
53 std::uniform_int_distribution<std::mt19937::result_type> dist(0, 25);
54
55 std::string res;
56 for (size_t i = 0; i < len; i++) {
57 res += 'a' + dist(rng);
58 }
59 return res;
60 }
61
62 public:
63 SpawnProc(std::string exeName_, bool captureOutput_ = false)
64 : exeName(exeName_), captureOutput(captureOutput_) {
65 auto dir = fs::path(TestContext::get().currentPath()).parent_path();
66 dir /= exeName;
67 exeName = dir.string();
68 if (captureOutput) {
69 auto path = fs::temp_directory_path();
70 path /= getRandomString();
71 tmpFileName = path.string();
72 }
73 }
74
75 int run() {
76 std::string execCmd = exeName;
77 if (captureOutput) {
78 execCmd += " > ";
79 execCmd += tmpFileName;
80 }
81
82 auto res = std::system(execCmd.c_str());
83
84 if (captureOutput) {
85 std::ifstream t(tmpFileName.c_str());
86 resultStr =
87 std::string((std::istreambuf_iterator<char>(t)), std::istreambuf_iterator<char>());
88 }
89 return res;
90 }
91
92 std::string getOutput() { return resultStr; }
93};
94} // namespace hip
