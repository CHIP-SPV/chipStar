128class OCLWrapperFunctions {
129protected:
130 // The map between the original kernel function and its wrapper
131 map<Function* , Function* > orig2Wrapper;

148 // The candidate kernel functions that need to be wrapped,
149 // 1. the spir kernel
150 // 2. the function arguments contain hipTexture type
151 bool collectCandidates() {
152 for (auto fi = M.begin(), fe = M.end(); fi != fe; fi ++) {
153 StringRef funcName = fi->getName();
154 Function* F = M.getFunction(funcName);
155
156 if (F->getCallingConv() == CallingConv::SPIR_KERNEL) {
157 // Check through function arguments
158 bool HasTextureType = false;
159 for (Function::arg_iterator ai = F->arg_begin(), ae = F->arg_end(); ai != ae; ai ++) {
160 if (textureTypes.IsHipTexturePtrTy(ai->getType())) {
161 HasTextureType = true;
162
163 break;
164 }
165 }
166
167 if (HasTextureType) {
168 orig2Wrapper[F] = nullptr;
169 }
170 } else if(F->getCallingConv() == CallingConv::SPIR_FUNC) {
171 // Check through function arguments
172 bool HasTextureType = false;
173 for (Function::arg_iterator ai = F->arg_begin(), ae = F->arg_end(); ai != ae; ai ++) {
174 if (textureTypes.IsHipTexturePtrTy(ai->getType())) {
175 HasTextureType = true;
176
177 break;
178 }
179 }
180
181 if (HasTextureType) {
182 spirFuncs.insert(F);
183 }
184 }
185 }
186
187 return orig2Wrapper.size() > 0;
188 }

190 // Generate the wrapper functions for the candidate kernels
191 bool wrapKernels() {
192 // Reset the attributes of spir functinos
193 for (Function* spirF : spirFuncs) {
194 // Reset the original function's attribute by eliminating the prohibition of optimizations
195 assignAttributes(spirF, spirF);
196 }
197
198 // Create all wrapper functions
199 for (map<Function* , Function* >::iterator wi = orig2Wrapper.begin(), we = orig2Wrapper.end();
200 wi != we; ++ wi) {
201 Function* F = wi->first;
202
203 // The map of argument ID from wrapper function to original function
204 SmallVector<int, 16> argMap;
205 // Create the wrapper function
206 Function* wrapperF = createWrapperFunction(F, argMap);
207
208 // Fix the original function regarding its texture argument's address space
209 F = fixTextureArgsAddrSpace(F, GENERIC_AS, SPIR_GLOBAL_AS);
210
211 // Fill in the content
212 fillContent(wrapperF, F, argMap);
213
214 // Register wrapper
215 wrapper2Orig[wrapperF] = F;
216
217 // Dump wrapper function
218 // wrapperF->dump();
219 }
220
221 return false;

225 // Create the wrapper function
226 Function* createWrapperFunction(Function* F, SmallVector<int, 16>& argMap) {
227 // Collect arugment types
228 SmallVector<Type *, 16> Parameters;
229 // The index mapping of the arguments from wrappe function to original function
230 int origIdx = 0;
231 int wrapperIdx = 0;
232 for (Function::const_arg_iterator ai = F->arg_begin(), ae = F->arg_end(); ai != ae; ++ ai) {
233 if (textureTypes.IsHipTexturePtrTy(ai->getType())) {
234 // Here we replace HipTexture type with OpenCL image and sampler types
235 Parameters.push_back(textureTypes.GetOCLImagePtrType(OCL_IMAGE_AS));
236 Parameters.push_back(textureTypes.GetOCLSamplerPtrType(OCL_SAMPLER_AS));
237
238 argMap.push_back(origIdx);
239 argMap.push_back(origIdx);
240 } else {
241 Parameters.push_back(ai->getType());
242
243 argMap.push_back(-1);
244 }
245
246 // Increment original function argument's ID
247 origIdx ++;
248 }
249
250 // Create the wrapper function
251 FunctionType * FuncTy = FunctionType::get(F->getReturnType(), Parameters, F->isVarArg());
252 Function * wrapperF = Function::Create(FuncTy, F->getLinkage(), F->getAddressSpace(), "", &M);
253 string wrapperName = "new_wrapper";
254 string origName = F->getName().str();
255 F->setName(origName + "_impl");
256
257 wrapperF->setName(origName);
258
259 // Inherit the attributes from original function
260 assignAttributes(wrapperF, F);
261
262 // Set visibility
263 wrapperF->setVisibility(GlobalValue::VisibilityTypes::HiddenVisibility);
264
265 // Set wrapper function's calling convention
266 wrapperF->setCallingConv(CallingConv::SPIR_KERNEL);
267
268 // Add wrapper function into module
269 M.getOrInsertFunction(wrapperF->getName(), wrapperF->getFunctionType(), wrapperF->getAttributes());
270
271 // Switch the oirignal function to implemmentation function by reset the name and calling convention
272 // Reset the calling convention to SPIR_FUNC
273 F->setCallingConv(CallingConv::SPIR_FUNC);
274
275 // Reset the original function's attribute by eliminating the prohibition of optimizations
276 assignAttributes(F, F);
277
278 return wrapperF;
279 }

347 bool initTextureStruct(Function* wrapperF, BasicBlock* contentBB, Instruction* textureAlloc,
348 Argument* imageArg, Argument* samplerArg) {
349 // Crete the type cast for image
350 Instruction* castImageInst = new PtrToIntInst(imageArg,
351 Type::getInt64Ty(M.getContext()),
352 "cast_image_to_int64ptr",
353 contentBB);
354
355 // Create the stroe related instruction
356 Value* imageIdxList[2] = {ConstantInt::get(Type::getInt32Ty(M.getContext()), 0),
357 ConstantInt::get(Type::getInt32Ty(M.getContext()), 0)};
358 Instruction* getImagePtrInst = GetElementPtrInst::Create(textureTypes.GetHipTextureType(),
359 textureAlloc,
360 ArrayRef<Value*>(imageIdxList, 2),
361 "image_field_address",
362 contentBB);
363
364 Instruction* storeImageInst = new StoreInst(castImageInst, getImagePtrInst, contentBB);
365
366 // Create the type cast for sampler
367 Instruction* castSamplerInst = new PtrToIntInst(samplerArg,
368 Type::getInt64Ty(M.getContext()),
369 "cast_sampler_to_int64ptr",
370 contentBB);
371
372 // Create the store related instruction
373 Value* samplerIdxList[2] = {ConstantInt::get(Type::getInt32Ty(M.getContext()), 0),
374 ConstantInt::get(Type::getInt32Ty(M.getContext()), 1)};
375 Instruction* getSamplerPtrInst = GetElementPtrInst::Create(textureTypes.GetHipTextureType(),
376 textureAlloc,
377 ArrayRef<Value*>(samplerIdxList, 2),
378 "sampler_field_address",
379 contentBB);

381 Instruction* storeSamplerInst = new StoreInst(castSamplerInst, getSamplerPtrInst, contentBB);

422 // Cast the address space
423 Instruction* castAddrSpace(Value* ptrVal, unsigned newAS, unsigned origAS, BasicBlock* contentBB) {
424 PointerType* ptrTy = dyn_cast<PointerType>(ptrVal->getType());
425 if (!ptrTy)
426 return nullptr;
427
428 Type* elemTy = ptrTy->getElementType();
429 if (origAS != ptrTy->getAddressSpace())
430 return nullptr;
431
432 Type* newTy = PointerType::get(elemTy, newAS);
433
434 return new AddrSpaceCastInst(ptrVal, newTy, "newASVal", contentBB);

437 // Fix the original function regarding its texture argument's address space
438 Function* fixTextureArgsAddrSpace(Function* origF, unsigned newAS, unsigned origAS) {
439 if (newAS != GENERIC_AS || origAS != SPIR_GLOBAL_AS)
440 return origF;
441
442 SmallVector<Type *, 16> ParamTys;
443 bool hasMutation = false;
444 for (Function::const_arg_iterator ai = origF->arg_begin(), ae = origF->arg_end(); ai != ae; ++ ai) {
445 Argument * arg = (Argument *)ai;
446
447 if (arg->getType() == textureTypes.GetHipTexturePtrType(origAS)) {
448 // Mutate argument type
449 arg->mutateType(textureTypes.GetHipTexturePtrType(newAS));
450
451 // Collect uses
452 SmallVector<Instruction *, 16> ascasts;
453 for (auto au : arg->users()) {
454 if (const AddrSpaceCastInst* ascast = dyn_cast<AddrSpaceCastInst>(au)) {
455 AddrSpaceCastInst* ascastInst = (AddrSpaceCastInst *)ascast;
456
457 // Record addresss cast instruction
458 ascasts.push_back(ascastInst);
459 hasMutation = true;
460 }
461 }
462
463 // Replace uses
464 for (Instruction* ascast : ascasts) {
465 ascast->replaceAllUsesWith(arg);
466 // Erase address cast instruction
467 ascast->eraseFromParent();
468 }
469
470 // Record new type
471 ParamTys.push_back(textureTypes.GetHipTexturePtrType(newAS));
472 } else {
473 // Record function argument types
474 ParamTys.push_back(arg->getType());
475 }
476 }
477
478 // Create the new function
479 if (hasMutation) {
480 FunctionType * FuncTy = FunctionType::get(origF->getReturnType(), ParamTys, origF->isVarArg());
481 Function *NewF = Function::Create(FuncTy, origF->getLinkage(), origF->getAddressSpace(), "", &M);
482 NewF->takeName(origF);
483
484 Function::arg_iterator AI = NewF->arg_begin();
485 ValueToValueMapTy VV;
486 for (Function::const_arg_iterator i = origF->arg_begin(), e = origF->arg_end(); i != e; ++i) {
487 AI->setName(i->getName());
488 VV[&*i] = &*AI;
489 ++ AI;
490 }
491
492 // Clone the function content
493 SmallVector<ReturnInst *, 1> RI;
494 CloneFunctionInto(NewF, origF, VV, CloneFunctionChangeType::GlobalChanges, RI);
495
496 // Reset the function attribute
497 assignAttributes(NewF, origF);
498
499 // Add new function into module
500 M.getOrInsertFunction(NewF->getName(), NewF->getFunctionType(), NewF->getAttributes());
501 // Erase original function
502 origF->eraseFromParent();
503
504 return NewF;
505 }
506
507 return origF;
508 }
