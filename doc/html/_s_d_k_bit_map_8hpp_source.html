229 void
230 load(const char * filename)
231 {
232 size_t val;
233 // Release any existing resources
234 releaseResources();
235 // Open BMP file
236 FILE * fd = fopen(filename, "rb");
237 // FILE *fd;
238 //fopen_s(&fd, filename, "rb");
239 // Opened OK
240 if (fd != NULL)
241 {
242 // Read header
243 val = fread((BitMapHeader *)this, sizeof(BitMapHeader), 1, fd);
244 // Failed to read header
245 if (ferror(fd))
246 {
247 fclose(fd);
248 return;
249 }
250 // Confirm that we have a bitmap file
251 if (id != bitMapID)
252 {
253 fclose(fd);
254 return;
255 }
256 // Read map info header
257 val = fread((BitMapInfoHeader *)this, sizeof(BitMapInfoHeader), 1, fd);
258 // Failed to read map info header
259 if (ferror(fd))
260 {
261 fclose(fd);
262 return;
263 }
264 // No support for compressed images
265 if (compression)
266 {
267 fclose(fd);
268 return;
269 }
270 // Support only 8 or 24 bits images
271 if (bitsPerPixel < 8)
272 {
273 fclose(fd);
274 return;
275 }
276 // Store number of colors
277 numColors_ = 1 << bitsPerPixel;
278 //load the palate for 8 bits per pixel
279 if(bitsPerPixel == 8)
280 {
281 colors_ = new ColorPalette[numColors_];
282 if (colors_ == NULL)
283 {
284 fclose(fd);
285 return;
286 }
287 val = fread(
288 (char *)colors_,
289 numColors_ * sizeof(ColorPalette),
290 1,
291 fd);
292 // Failed to read colors
293 if (ferror(fd))
294 {
295 fclose(fd);
296 return;
297 }
298 }
299 // Allocate buffer to hold all pixels
300 unsigned int sizeBuffer = size - offset;
301 unsigned char * tmpPixels = new unsigned char[sizeBuffer];
302 if (tmpPixels == NULL)
303 {
304 delete colors_;
305 colors_ = NULL;
306 fclose(fd);
307 return;
308 }
309 // Read pixels from file, including any padding
310 val = fread(tmpPixels, sizeBuffer * sizeof(unsigned char), 1, fd);
311 // Failed to read pixel data
312 if (ferror(fd))
313 {
314 delete colors_;
315 colors_ = NULL;
316 delete [] tmpPixels;
317 fclose(fd);
318 return;
319 }
320 // Allocate image
321 pixels_ = new uchar4[width * height];
322 if (pixels_ == NULL)
323 {
324 delete colors_;
325 colors_ = NULL;
326 delete [] tmpPixels;
327 fclose(fd);
328 return;
329 }
330 // Set image, including w component (white)
331 memset(pixels_, 0xff, width * height * sizeof(uchar4));
332 unsigned int index = 0;
333 for(int y = 0; y < height; y++)
334 {
335 for(int x = 0; x < width; x++)
336 {
337 // Read RGB values
338 if (bitsPerPixel == 8)
339 {
340 pixels_[(y * width + x)] = colors_[tmpPixels[index++]];
341 }
342 else // 24 bit
343 {
344 pixels_[(y * width + x)].z = tmpPixels[index++];
345 pixels_[(y * width + x)].y = tmpPixels[index++];
346 pixels_[(y * width + x)].x = tmpPixels[index++];
347 }
348 }
349 // Handle padding
350 for(int x = 0; x < (4 - (3 * width) % 4) % 4; x++)
351 {
352 index++;
353 }
354 }
355 // Loaded file so we can close the file.
356 fclose(fd);
357 delete[] tmpPixels;
358 // Loaded file so record this fact
359 isLoaded_ = true;
360 }
361 }

373 bool
374 write(const char * filename)
375 {
376 if (!isLoaded_)
377 {
378 return false;
379 }
380 // Open BMP file
381 FILE * fd = fopen(filename, "wb");
382 //FILE * fd;
383 //fopen_s(&fd, filename, "wb");
384 // Opened OK
385 if (fd != NULL)
386 {
387 // Write header
388 fwrite((BitMapHeader *)this, sizeof(BitMapHeader), 1, fd);
389 // Failed to write header
390 if (ferror(fd))
391 {
392 fclose(fd);
393 return false;
394 }
395 // Write map info header
396 fwrite((BitMapInfoHeader *)this, sizeof(BitMapInfoHeader), 1, fd);
397 // Failed to write map info header
398 if (ferror(fd))
399 {
400 fclose(fd);
401 return false;
402 }
403 // Write palate for 8 bits per pixel
404 if(bitsPerPixel == 8)
405 {
406 fwrite(
407 (char *)colors_,
408 numColors_ * sizeof(ColorPalette),
409 1,
410 fd);
411 // Failed to write colors
412 if (ferror(fd))
413 {
414 fclose(fd);
415 return false;
416 }
417 }
418 for(int y = 0; y < height; y++)
419 {
420 for(int x = 0; x < width; x++)
421 {
422 // Read RGB values
423 if (bitsPerPixel == 8)
424 {
425 fputc(
426 colorIndex(
427 pixels_[(y * width + x)]),
428 fd);
429 }
430 else // 24 bit
431 {
432 fputc(pixels_[(y * width + x)].z, fd);
433 fputc(pixels_[(y * width + x)].y, fd);
434 fputc(pixels_[(y * width + x)].x, fd);
435 if (ferror(fd))
436 {
437 fclose(fd);
438 return false;
439 }
440 }
441 }
442 // Add padding
443 for(int x = 0; x < (4 - (3 * width) % 4) % 4; x++)
444 {
445 fputc(0, fd);
446 }
447 }
448 return true;
449 }
450 return false;

453 bool
454 write(const char * filename, int width, int height, unsigned int *ptr)
455 {
456 // Open BMP file
457 FILE * fd = fopen(filename, "wb");
458 int alignSize = width * 4;
459 alignSize ^= 0x03;
460 alignSize ++;
461 alignSize &= 0x03;
462 int rowLength = width * 4 + alignSize;
463 // Opened OK
464 if (fd != NULL)
465 {
466 BitMapHeader *bitMapHeader = new BitMapHeader;
467 bitMapHeader->id = bitMapID;
468 bitMapHeader->offset = sizeof(BitMapHeader) + sizeof(BitMapInfoHeader);
469 bitMapHeader->reserved1 = 0x0000;
470 bitMapHeader->reserved2 = 0x0000;
471 bitMapHeader->size = sizeof(BitMapHeader) + sizeof(BitMapInfoHeader) + rowLength
472 * height;
473 // Write header
474 fwrite(bitMapHeader, sizeof(BitMapHeader), 1, fd);
475 // Failed to write header
476 if (ferror(fd))
477 {
478 fclose(fd);
479 return false;
480 }
481 BitMapInfoHeader *bitMapInfoHeader = new BitMapInfoHeader;
482 bitMapInfoHeader->bitsPerPixel = 32;
483 bitMapInfoHeader->clrImportant = 0;
484 bitMapInfoHeader->clrUsed = 0;
485 bitMapInfoHeader->compression = 0;
486 bitMapInfoHeader->height = height;
487 bitMapInfoHeader->imageSize = rowLength * height;
488 bitMapInfoHeader->planes = 1;
489 bitMapInfoHeader->sizeInfo = sizeof(BitMapInfoHeader);
490 bitMapInfoHeader->width = width;
491 bitMapInfoHeader->xPelsPerMeter = 0;
492 bitMapInfoHeader->yPelsPerMeter = 0;
493 // Write map info header
494 fwrite(bitMapInfoHeader, sizeof(BitMapInfoHeader), 1, fd);
495 // Failed to write map info header
496 if (ferror(fd))
497 {
498 fclose(fd);
499 return false;
500 }
501 unsigned char buffer[4];
502 int x, y;
503 for (y = 0; y < height; y++)
504 {
505 for (x = 0; x < width; x++, ptr++)
506 {
507 if( 4 != fwrite(ptr, 1, 4, fd))
508 {
509 fclose(fd);
510 return false;
511 }
512 }
513 memset( buffer, 0x00, 4 );
514 fwrite( buffer, 1, alignSize, fd );
515 }
516 fclose( fd );
517 return true;
518 }
519 return false;
520 }
