335 float getElapsedTime(CHIPEvent* other_) override {
336 CHIPEventLevel0* other = (CHIPEventLevel0*)other_;
337 // std::lock_guard<std::mutex> Lock(ContextMutex);
338
339 if (!this->isRecordingOrRecorded() || !other->isRecordingOrRecorded())

341
342 this->updateFinishStatus();
343 other->updateFinishStatus();
344 if (!this->isFinished() || !other->isFinished()) return hipErrorNotReady;
345
346 uint64_t Started = this->getFinishTime();
347 uint64_t Finished = other->getFinishTime();
348 if (Started > Finished) std::swap(Started, Finished);
349
350 CHIPContextLevel0* chip_ctx_lz = (CHIPContextLevel0*)chip_context;
351 CHIPDeviceLevel0* chip_dev_lz =
352 (CHIPDeviceLevel0*)chip_ctx_lz->getDevices()[0];
353
354 auto props = chip_dev_lz->getDeviceProps();
355
356 uint64_t timerResolution = props->timerResolution;
357 uint32_t timestampValidBits = props->timestampValidBits;
358
359 logDebug(
360 "EventElapsedTime: Started {} Finished {} timerResolution {} "
361 "timestampValidBits {}\n",
362 Started, Finished, timerResolution, timestampValidBits);
363
364 Started = (Started & (((uint64_t)1 << timestampValidBits) - 1));
365 Finished = (Finished & (((uint64_t)1 << timestampValidBits) - 1));
366 if (Started > Finished)
367 Finished = Finished + ((uint64_t)1 << timestampValidBits) - Started;
368 Started *= timerResolution;
369 Finished *= timerResolution;
370
371 logDebug("EventElapsedTime: STARTED {} FINISHED {} \n", Started, Finished);
372
373 // apparently fails for Intel NEO, god knows why
374 // assert(Finished >= Started);
375 uint64_t Elapsed;
376#define NANOSECS 1000000000
377 uint64_t MS = (Elapsed / NANOSECS) * 1000;
378 uint64_t NS = Elapsed % NANOSECS;
379 float FractInMS = ((float)NS) / 1000000.0f;
380 auto ms = (float)MS + FractInMS;
381
382 return ms;
